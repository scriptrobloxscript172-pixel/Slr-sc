local function create(class, props)
    local inst = Instance.new(class)
    for k, v in pairs(props) do
        if k ~= "Parent" then
            inst[k] = v
        end
    end
    if props.Parent then inst.Parent = props.Parent end
    return inst
end

local TweenService = game:GetService("TweenService")

local function CreateStreak()
	local Line = Instance.new("Frame")
	Line.Parent = ParticleHolder
	Line.AnchorPoint = Vector2.new(0.5,0.5)
	Line.Size = UDim2.new(0,120,0,3)
	Line.Position = UDim2.new(-0.2,0,math.random(),0)
	Line.BackgroundColor3 = Color3.fromRGB(255,0,255)
	Line.BorderSizePixel = 0
	Line.Rotation = -25
	Line.ZIndex = 0

	local Corner = Instance.new("UICorner")
	Corner.CornerRadius = UDim.new(1,0)
	Corner.Parent = Line

	local Glow = Instance.new("UIStroke")
	Glow.Parent = Line
	Glow.Color = Color3.fromRGB(255,0,255)
	Glow.Thickness = 2

	local Tween = TweenService:Create(
		Line,
		TweenInfo.new(2.5, Enum.EasingStyle.Linear),
		{
			Position = UDim2.new(1.2,0,math.random(),0),
			BackgroundTransparency = 1
		}
	)

	Tween:Play()

	Tween.Completed:Connect(function()
		Line:Destroy()
	end)
end

            function parentObj:Slider(sName, min, max, default, callback)
                local sliderVal = default or min
                local slFrame = create("Frame", {
                    Parent = parentFrame,
                    Size = UDim2.new(1, -10, 0, 45),
                    BackgroundColor3 = Color3.fromRGB(30, 30, 30),

                local slLabel = create("TextLabel", {
                    Parent = slFrame,
                    Size = UDim2.new(1, -20, 0, 20),
                    Position = UDim2.new(0, 10, 0, 4),
                    BackgroundTransparency = 1,
                    Text = sName .. " : " .. tostring(sliderVal),
                    TextColor3 = Color3.fromRGB(255, 255, 255),
                    Font = Enum.Font.Gotham,
                    TextSize = 12,
                    TextXAlignment = Enum.TextXAlignment.Left
                })

                local slBar = create("TextButton", {
                    Parent = slFrame,
                    Size = UDim2.new(1, -20, 0, 4),
                    Position = UDim2.new(0, 10, 0, 30),
                    BackgroundColor3 = Color3.fromRGB(50, 50, 50),
                    Text = ""
                })
                create("UICorner", {Parent = slBar, CornerRadius = UDim.new(1, 0)})

                local slFill = create("Frame", {
                    Parent = slBar,
                    Size = UDim2.new((sliderVal - min) / (max - min), 0, 1, 0),
                    BackgroundColor3 = Color3.fromRGB(170, 0, 255)
                })
                create("UICorner", {Parent = slFill, CornerRadius = UDim.new(1, 0)})

                local draggingSl = false
                local function updateSl(input)
                    local pos = math.clamp((input.Position.X - slBar.AbsolutePosition.X) / slBar.AbsoluteSize.X, 0, 1)
                    -- Fix to support decimal sliders
                    local isFloat = (math.floor(min) ~= min or math.floor(max) ~= max)
                    if isFloat then
                        sliderVal = tonumber(string.format("%.1f", min + (max - min) * pos))
                    else
                        sliderVal = math.floor(min + (max - min) * pos)
                    end
                    
                    slLabel.Text = sName .. " : " .. tostring(sliderVal)
                    ts:Create(slFill, TweenInfo.new(0.1), {Size = UDim2.new(pos, 0, 1, 0)}):Play()
                    if callback then callback(sliderVal) end
                end

                slBar.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        draggingSl = true
                        updateSl(input)
                    end
                end)
                uis.InputEnded:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        draggingSl = false
                    end
                end)
                uis.InputChanged:Connect(function(input)
                    if draggingSl and input.UserInputType == Enum.UserInputType.MouseMovement then
                        updateSl(input)
                    end
                end)
            end

local function isVisible(targetPart)
    if not targetPart then return false end
    local origin = cam.CFrame.Position
    local direction = (targetPart.Position - origin).Unit * (targetPart.Position - origin).Magnitude
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = {lp.Character, cam}
    params.FilterType = Enum.RaycastFilterType.Blacklist
    local result = ws:Raycast(origin, direction, params)
    return result and result.Instance:IsDescendantOf(targetPart.Parent) or not result
end

local function getTarget()
    local target = nil
    local shortestDist = C.FovSize
    for _, v in pairs(plrs:GetPlayers()) do
        if v ~= lp and v.Character and v.Character:FindFirstChild("Humanoid") then
            if C.AliveCheck and v.Character.Humanoid.Health <= 0 then continue end
            local part = v.Character:FindFirstChild(C.AimBone) or v.Character:FindFirstChild("Head")
            if part then
                local pos, onScreen = cam:WorldToViewportPoint(part.Position)
                if onScreen then
                    if C.VisCheck and not isVisible(part) then continue end
                    local dist = (Vector2.new(mouse.X, mouse.Y) - Vector2.new(pos.X, pos.Y)).Magnitude
                    if dist < shortestDist then
                        target = part
                        shortestDist = dist
                    end
                end
            end
        end
    end
    return target
end

local function createEsp(player)
    local esp = {
        box = Drawing.new("Square"), name = Drawing.new("Text"), health = Drawing.new("Text"),
        dist = Drawing.new("Text"), tracer = Drawing.new("Line"), skeleton = {}
    }
    esp.box.Color = Color3.fromRGB(255, 255, 255); esp.box.Thickness = 1; esp.box.Filled = false
    esp.name.Color = Color3.fromRGB(255, 255, 255); esp.name.Size = 16; esp.name.Center = true; esp.name.Outline = true
    esp.health.Size = 14; esp.health.Center = true; esp.health.Outline = true
    esp.dist.Color = Color3.fromRGB(200, 200, 200); esp.dist.Size = 14; esp.dist.Center = true; esp.dist.Outline = true
    esp.tracer.Color = Color3.fromRGB(255, 255, 255); esp.tracer.Thickness = 1
    for i = 1, #boneConnections do
        local line = Drawing.new("Line")
        line.Color = Color3.fromRGB(255, 255, 255); line.Thickness = 1
        esp.skeleton[i] = line
    end
    espCache[player] = esp
end

local function removeEsp(player)
    if espCache[player] then
        for k, v in pairs(espCache[player]) do
            if k == "skeleton" then
                for _, line in ipairs(v) do line:Remove() end
            else
                v:Remove()
            end
        end
        espCache[player] = nil
    end
end

rs.RenderStepped:Connect(function()
    -- CFrame Fly (Player)
    if C.Fly then
        local hrp = lp.Character and lp.Character:FindFirstChild("HumanoidRootPart")
        if hrp then
            hrp.Velocity = Vector3.new(0, 0, 0)
            local camCFrame = cam.CFrame
            local moveDir = Vector3.new()
            if flyCtrl.f + flyCtrl.b ~= 0 then moveDir = moveDir + (camCFrame.LookVector * (flyCtrl.f + flyCtrl.b)) end
            if flyCtrl.l + flyCtrl.r ~= 0 then moveDir = moveDir + (camCFrame.RightVector * (flyCtrl.l + flyCtrl.r)) end
            if flyCtrl.u + flyCtrl.d ~= 0 then moveDir = moveDir + (camCFrame.UpVector * (flyCtrl.u + flyCtrl.d)) end
            if moveDir.Magnitude > 0 then moveDir = moveDir.Unit end
            hrp.CFrame = hrp.CFrame + (moveDir * (C.FlySpeed / 50))
        end
    end

    -- Car Fly Logic
    if C.CarFly then
        local char = lp.Character
        if char and char:FindFirstChild("Humanoid") and char.Humanoid.SeatPart and char.Humanoid.SeatPart:IsA("VehicleSeat") then
            local seat = char.Humanoid.SeatPart
            local camCFrame = cam.CFrame
            local moveDir = Vector3.new()
            if flyCtrl.f + flyCtrl.b ~= 0 then moveDir = moveDir + (camCFrame.LookVector * (flyCtrl.f + flyCtrl.b)) end
            if flyCtrl.l + flyCtrl.r ~= 0 then moveDir = moveDir + (camCFrame.RightVector * (flyCtrl.l + flyCtrl.r)) end
            if moveDir.Magnitude > 0 then moveDir = moveDir.Unit end
            
            -- Override Assembly Velocity for flight
            seat.AssemblyLinearVelocity = moveDir * C.CarFlySpeed
            seat.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
        end
    end

    if C.Fov then
        fovCircle.Position = Vector2.new(mouse.X, mouse.Y + 36)
        fovCircle.Radius = C.FovSize
        fovCircle.Visible = true
    else
        fovCircle.Visible = false
    end

    if C.Aim_Camera then
        local target = getTarget()
        if target then
            cam.CFrame = cam.CFrame:Lerp(CFrame.new(cam.CFrame.Position, target.Position), 0.5)
        end
    end

    for player, esp in pairs(espCache) do
        local hasChar = player.Character and player.Character:FindFirstChild("HumanoidRootPart") and player.Character:FindFirstChild("Humanoid")
        if C.Esp and hasChar and player.Character.Humanoid.Health > 0 then
            local hrp = player.Character.HumanoidRootPart
            local pos, onScreen = cam:WorldToViewportPoint(hrp.Position)
            
            if onScreen then
                local size = Vector3.new(2, 3, 0) * (cam.ViewportSize.Y / pos.Z)
                local tlc = Vector2.new(pos.X - size.X / 2, pos.Y - size.Y / 2)
                local distance = math.floor((cam.CFrame.Position - hrp.Position).Magnitude)
                local hp = math.floor(player.Character.Humanoid.Health)
                local maxHp = math.floor(player.Character.Humanoid.MaxHealth)
                
                esp.box.Visible = C.Box; esp.box.Size = Vector2.new(size.X, size.Y); esp.box.Position = tlc
                esp.name.Visible = C.Name; esp.name.Position = Vector2.new(pos.X, tlc.Y - 18); esp.name.Text = player.Name
                esp.health.Visible = C.Health; esp.health.Position = Vector2.new(pos.X, tlc.Y + size.Y + 2)
                esp.health.Text = "HP: " .. tostring(hp) .. "/" .. tostring(maxHp)
                esp.health.Color = Color3.fromRGB(255 - (hp/maxHp)*255, (hp/maxHp)*255, 0)
                esp.dist.Visible = C.Dist; esp.dist.Position = Vector2.new(pos.X, tlc.Y + size.Y + (C.Health and 16 or 2))
                esp.dist.Text = "[" .. tostring(distance) .. "m]"
                esp.tracer.Visible = C.Tracer; esp.tracer.From = Vector2.new(cam.ViewportSize.X / 2, cam.ViewportSize.Y); esp.tracer.To = Vector2.new(pos.X, tlc.Y + size.Y)
                
                if C.Skeleton then
                    for i, connection in ipairs(boneConnections) do
                        local p1 = player.Character:FindFirstChild(connection[1])
                        local p2 = player.Character:FindFirstChild(connection[2])
                        if p1 and p2 then
                            local pos1, vis1 = cam:WorldToViewportPoint(p1.Position)
                            local pos2, vis2 = cam:WorldToViewportPoint(p2.Position)
                            if vis1 and vis2 then
                                esp.skeleton[i].Visible = true; esp.skeleton[i].From = Vector2.new(pos1.X, pos1.Y); esp.skeleton[i].To = Vector2.new(pos2.X, pos2.Y)
                            else
                                esp.skeleton[i].Visible = false
                            end
                        else
                            esp.skeleton[i].Visible = false
                        end
                    end
                else
                    for _, line in ipairs(esp.skeleton) do line.Visible = false end
                end
            else
                esp.box.Visible = false; esp.name.Visible = false; esp.health.Visible = false; esp.dist.Visible = false; esp.tracer.Visible = false
                for _, line in ipairs(esp.skeleton) do line.Visible = false end
            end
        else
            esp.box.Visible = false; esp.name.Visible = false; esp.health.Visible = false; esp.dist.Visible = false; esp.tracer.Visible = false
            for _, line in ipairs(esp.skeleton) do line.Visible = false end
        end
    end
end)

local CarsTab = Window:Tab("Cars")

local function GetActiveVehicles()
    local vehicles = {}
    local added = {}
    for _, obj in pairs(ws:GetDescendants()) do
        if obj:IsA("VehicleSeat") and obj.Parent then
            local carName = obj.Parent.Name
            if not added[carName] then
                table.insert(vehicles, carName)
                added[carName] = true
            end
        end
    end
    if #vehicles == 0 then table.insert(vehicles, "None") end
    return vehicles
end

local function GetCurrentVehicle()
    local char = lp.Character
    if char and char:FindFirstChild("Humanoid") and char.Humanoid.SeatPart then
        if char.Humanoid.SeatPart:IsA("VehicleSeat") then
            return char.Humanoid.SeatPart.Parent, char.Humanoid.SeatPart
        end
    end
    return nil, nil
end

local function ApplyCarMod(modName, value)
    if not C.CarMods then return end
    local car = GetCurrentVehicle()
    if not car then return end
    
    local tuneFolder = car:FindFirstChild("Tune", true) or car:FindFirstChild("Handling", true)
    if tuneFolder then
        for _, v in pairs(tuneFolder:GetDescendants()) do
            if v:IsA("NumberValue") and string.find(string.lower(v.Name), string.lower(modName)) then
                v.Value = v.Value * value
            end
        end
    end
end

local ModSliders = {
    "Acceleration", "Brake Force", "Max Speed", "Steering Speed", 
    "Turbo Boost", "Redline RPM", "Gearing", "Suspension",

    "Tire Grip", "Peak Torque"
}

for _, modName in ipairs(ModSliders) do
    CarModsSec:Slider(modName, 1, 10, 1, function(Value)
        ApplyCarMod(modName, Value)
    end)
end

                    if fuel and fuel:IsA("NumberValue") then fuel.Value = 100 end
                end)

            if C.InfCarHealth then
                pcall(function()
                    local hp = car:FindFirstChild("Health", true) or car:FindFirstChild("CarHealth", true)

                    if hp and hp:IsA("NumberValue") then hp.Value = maxHp and maxHp.Value or 1000 end
                end)
            end
        end
    end
end)
